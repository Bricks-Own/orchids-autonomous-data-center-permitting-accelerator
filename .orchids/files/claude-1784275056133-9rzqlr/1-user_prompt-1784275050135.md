import React from 'react';

import { AIR_MODULES, STATES_ATTAINMENT } from '../data/permitData';

import { THRESHOLDS } from '../utils/calculations';

// ── Clickable Pathway Badge (with hover tooltip) ──

function PathwayBadge({ label, status, detail }) {

  const [showTip, setShowTip] = React.useState(false);

  const colorMap = {

    warning: 'bg-destructive/10 border-red-700/40 text-destructive',

    ok: 'bg-primary/10 border-green-700/40 text-primary',

    neutral: 'bg-muted border-border text-muted-foreground',

  };

  const IconMap = { warning: XCircle, ok: CheckCircle, neutral: WarningCircle };

  const Icon = IconMap[status] || WarningCircle;

  return (

    <div className="relative">

      <button

        onMouseEnter={() => setShowTip(true)}

        onMouseLeave={() => setShowTip(false)}

        onClick={() => setShowTip(!showTip)}

        className={`text-xs px-2.5 py-1 border transition-colors ${colorMap[status] || colorMap.neutral} cursor-pointer hover:opacity-80 inline-flex items-center gap-1.5`}

      >

        <Icon weight="fill" size={12} />

        {label}

      </button>

      {showTip && (

        <div className="absolute z-20 top-full mt-1.5 right-0 w-64 bg-muted border border-border p-3 shadow-black/40">

          <div className="text-xs text-foreground/80 leading-relaxed">{detail}</div>

          <div className="absolute -top-1.5 right-4 w-3 h-3 bg-muted border-t border-l border-border rotate-45" />

        </div>

      )}

    </div>

  );

}

export default function AirPermitAI({ results, inputs, setActiveTab }) {

  const c = PERMIT_COLORS.air;

  if (!results) {

    return (

      <div className="p-6 text-center py-20">

        <Wind weight="duotone" size={48} className="mx-auto text-muted-foreground mb-4" />

        <h3 className="text-muted-foreground font-medium mb-2">No Site Data Yet</h3>

        <p className="text-muted-foreground/70 text-sm mb-4">Go to Site Intake and run the permit screening first.</p>

        <Button variant="default" onClick={() => setActiveTab && setActiveTab('intake')}>

          Open Site Intake

        </Button>

      </div>

    );

  }

  const { baseline, controlled, pathway, totalMW, breaches } = results;

  const attainment = STATES_ATTAINMENT[inputs.state] || '';

  const isNonAttain = attainment.includes('Nonattainment') || inputs.nonAttainment;

  const breachedItems = breaches?.filter(b => b.status === 'BREACHED') || [];

  const mitigatedItems = breaches?.filter(b => b.status === 'MITIGATED') || [];

  const noticeItems = breaches?.filter(b => b.status === 'NOTICE') || [];

  const hasBreaches = breaches && breaches.length > 0;

  // ── Metric cards ──

  const airMetrics = [

    {

      label: 'Baseline NOx (Uncontrolled)',

      value: `${baseline.nox?.toFixed(1) || '—'} tpy`,

      sub: `PSD threshold: ${THRESHOLDS.nox.psd} tpy`,

      accent: 'text-destructive',

    },

    {

      label: 'Brick-Controlled NOx',

      value: `${controlled.nox?.toFixed(1) || '—'} tpy`,

      sub: `${((controlled.nox / (baseline.nox || 1)) * 100).toFixed(0)}% of baseline`,

      accent: c.accent,

    },

    {

      label: 'CO₂e (GHGRP)',

      value: `${baseline.co2e?.toFixed(0) || '—'} tpy`,

      sub: `Threshold: ${THRESHOLDS.co2e.ghgrp.toLocaleString()} tpy`,

      accent: baseline.co2e >= THRESHOLDS.co2e.ghgrp ? 'text-destructive' : 'text-primary',

    },

    {

      label: 'Threshold Status',

      value: breachedItems.length > 0 ? `${breachedItems.length} Breached` : 'All Clear',

      sub: breachedItems.length > 0

        ? `${mitigatedItems.length} mitigated · ${noticeItems.length} notices`

        : `${mitigatedItems.length} mitigated · No exceedances`,

      accent: breachedItems.length > 0 ? 'text-destructive' : 'text-primary',

    },

  ];

  // ── Banner badges ──

  const bannerBadges = [

    { label: pathway.requiresPSD ? 'PSD Major Source' : 'Below PSD', status: pathway.requiresPSD ? 'warning' : 'ok', detail: pathway.requiresPSD ? 'One or more criteria pollutants ≥100 tpy' : 'All criteria pollutants below 100 tpy PSD threshold' },

    { label: pathway.requiresTitleV ? 'Title V Required' : 'Below Title V', status: pathway.requiresTitleV ? 'warning' : 'ok', detail: pathway.requiresTitleV ? 'Controlled emissions ≥100 tpy triggers Title V' : 'All controlled emissions below 100 tpy Title V threshold' },

    { label: pathway.syntheticMinorViable ? 'Synthetic Minor Viable' : 'Synth. Minor Not Viable', status: pathway.syntheticMinorViable ? 'ok' : 'warning', detail: pathway.syntheticMinorViable ? 'Brick controls reduce all pollutants below 100 tpy' : 'At least one pollutant cannot be reduced below major source thresholds' },

  ];

  // ── Pathway items ──

  const psdBreach = breachedItems.find(b => b.thresholdType === 'PSD Major Source');

  const nonAttainBreach = breachedItems.find(b => b.thresholdType === 'Nonattainment NSR');

  const synthMinorBreach = breachedItems.find(b => b.id === 'synthetic_minor_viability');

  // Build a detail string from a breach object

  function breachDetail(breach) {

    if (!breach) return '';

    let d = breach.description;

    if ((breach.baseline > 0 || breach.controlled > 0) && breach.threshold > 0) {

      d += `\n\nBaseline PTE: ${breach.baseline.toFixed(1)} ${breach.unit}  |  Brick-Controlled: ${breach.controlled.toFixed(1)} ${breach.unit}  |  Threshold: ${breach.threshold.toFixed(1)} ${breach.unit}.`;

      if (breach.margin !== undefined) {

        d += breach.margin >= 0

          ? `\nMargin: +${breach.margin.toFixed(1)} ${breach.unit} below threshold — within compliance.`

          : `\nMargin: ${Math.abs(breach.margin).toFixed(1)} ${breach.unit} over threshold — corrective action needed.`;

      }

    }

    if (breach.remediationSteps && breach.remediationSteps.length > 0) {

      d += '\n\nRemediation Plan:';

      breach.remediationSteps.forEach(step => {

        d += `\nStep ${step.stepNumber}: ${step.title} — ${step.description}`;

        if (step.techOptions && step.techOptions.length > 0) {

          d += '\nTechnology options:';

          step.techOptions.forEach(opt => {

            d += `\n  - ${opt.name}: ${opt.reduction} reduction, ${opt.timeline}, ${opt.cost}, ${opt.complexity} complexity`;

          });

        }

      });

    }

    if (breach.brickControl && !breach.remediationSteps) d += `\n\nBrick Control Applied: ${breach.brickControl}`;

    if (breach.additionalControls && !breach.remediationSteps) d += `\n\nAdditional Controls Required: ${breach.additionalControls}`;

    return d;

  }

  const pathwayItems = [];

  // All-clear item when no breaches

  if (!hasBreaches) {

    pathwayItems.push({

      label: 'All Thresholds Clear',

      status: 'Compliant',

      ok: true,

      detail: 'All pollutant emissions are below applicable PSD, Title V, and regulatory thresholds. No breaches detected.',

      trigger: 'No corrective action needed',

    });

  }

  pathwayItems.push({

    label: 'NSR/PSD Applicability',

    status: pathway.requiresPSD ? 'PSD Required (>=100 tpy)' : 'Below PSD threshold',

    ok: !pathway.requiresPSD,

    detail: `CAA ss 165 / 40 CFR Parts 51/52. PSD applies to new major sources in attainment areas.${psdBreach ? '\n\n' + breachDetail(psdBreach) : ''}\n\nIf PSD applies, full BACT analysis is required. Recommended BACT technologies:\n• Dry Low NOx (DLN) Combustion — ≤9–15 ppmvd @ 15% O₂ (Step 1 — Available)\n• Selective Catalytic Reduction (SCR) — ≤2–5 ppmvd @ 15% O₂ (Step 2 — Evaluate for breaches)\n• Oxidation Catalyst — CO reduction 90%+ (Step 2 — Often Required)\n• Operational Limits (Brick Controls) — Runtime cap + load optimization (Step 3)\n• Fuel Switching / Natural Gas Only — Eliminates SO₂/PM\n• Good Combustion Practices (GCP) — Air/fuel ratio, maintenance tuning`,

    breach: psdBreach,

    regText: '40 CFR ss 52.21 / CAA ss 165',

  });

  pathwayItems.push({

    label: 'Nonattainment NSR',

    status: isNonAttain ? 'Nonattainment NSR may apply' : 'Not triggered (attainment area)',

    ok: !isNonAttain,

    detail: `CAA ss 173. If county is nonattainment, NSR/LAER offsets required at 25 tpy threshold. Offsets ratio of at least 1.3:1 required for severe nonattainment areas.${nonAttainBreach ? '\n\n' + breachDetail(nonAttainBreach) : ''}`,

    breach: nonAttainBreach,

    regText: '40 CFR ss 51.165 / CAA ss 173',

  });

  pathwayItems.push({

    label: 'Title V Operating Permit',

    status: pathway.requiresTitleV ? 'Title V Required (aggregate major source)' : 'Below Title V thresholds',

    ok: !pathway.requiresTitleV,

    detail: '40 CFR Part 70/71. Required if PTE >= 100 tpy for any regulated air pollutant. Title V permit includes compliance certification, monitoring schedules, and annual fee program.',

    regText: '40 CFR Part 70/71',

  });

  pathwayItems.push({

    label: 'Synthetic Minor Pathway',

    status: pathway.syntheticMinorViable ? `Viable — Brick controls reduce PTE to ${controlled.nox?.toFixed(1)} tpy NOx` : 'Not viable',

    ok: pathway.syntheticMinorViable,

    detail: `Enforceable operational limits (runtime caps, fuel use limits, dispatch optimization) can reduce PTE below major source thresholds, avoiding PSD. Requires state-issued synthetic minor permit with federally enforceable conditions.${synthMinorBreach ? '\n\n' + breachDetail(synthMinorBreach) : ''}`,

    breach: synthMinorBreach,

    regText: '40 CFR ss 51.165 / 51.166',

  });

  pathwayItems.push({

    label: 'NSPS — 40 CFR Part 60 Subpart KKKK/KKKKa',

    status: 'Applicable — Stationary combustion turbines',

    ok: false,

    detail: `New/modified turbines >=1 MW: NOx limit ${THRESHOLDS.nox.nspsLimit} ppmvd, CO limit ${THRESHOLDS.co.nspsLimit} ppmvd @ 15% O2. Initial performance stack test within 180 days of startup + ongoing compliance.\n\nBACT Technology Reference:\nStep 1 (Available): Dry Low NOx (DLN) Combustion — ≤9–15 ppmvd @ 15% O₂, combustion controls.\nStep 2 (Evaluate): SCR — ≤2–5 ppmvd @ 15% O₂, post-combustion, 90%+ reduction.\nStep 2 (Often Required): Oxidation Catalyst — CO reduction 90%+, post-combustion.\nStep 3 (Enforceable): Operational Limits — dispatch optimization, runtime cap.`,

    regText: '40 CFR Part 60 Subparts KKKK/KKKKa',

  });

  pathwayItems.push({

    label: 'NESHAP — 40 CFR Part 63 Subpart YYYY',

    status: baseline.hap >= THRESHOLDS.hap.combined ? 'Major HAP source — Subpart YYYY applies' :

            baseline.hap >= THRESHOLDS.hap.single ? 'Near major HAP — confirm applicability' : 'Area source — confirm',

    ok: baseline.hap < THRESHOLDS.hap.single,

    detail: 'Applies at major HAP sources (>=10 tpy single HAP or >=25 tpy combined HAP). MACT standards for gas turbines. If area source, still subject to GCP standards.',

    regText: '40 CFR Part 63 Subparts YYYY/ZZZZ',

  });

  pathwayItems.push({

    label: 'Engine NSPS/NESHAP (Subparts IIII/JJJJ/ZZZZ)',

    status: `${inputs.gensetCount} backup gensets at ${inputs.gensetHours} hr/yr — ${inputs.gensetHours <= 100 ? 'within emergency limit' : 'EXCEEDS 100 hr/yr limit'}`,

    ok: inputs.gensetHours <= 100,

    detail: '40 CFR Parts 60 IIII/JJJJ and 63 ZZZZ. Emergency engines limited to 100 hr/yr maintenance + emergency. Non-resettable hour meters required. Tier 4 compliant engines for new installations.',

    regText: '40 CFR 60 IIII/JJJJ, 63 ZZZZ',

  });

  pathwayItems.push({

    label: 'GHG / GHGRP (40 CFR Part 98)',

    status: baseline.co2e >= THRESHOLDS.co2e.ghgrp ? 'GHGRP Required (>=25,000 tpy CO2e)' : 'Below GHGRP threshold (25,000 tpy)',

    ok: baseline.co2e < THRESHOLDS.co2e.ghgrp,

    detail: 'Annual GHG report to EPA eGGRT required if >=25,000 tpy CO2e. Subpart C (stationary combustion) + Subpart W (fugitives). Third-party verification required for >25,000 tpy.',

    regText: '40 CFR Part 98 Subparts C/W',

  });

  return (

    <div className="p-6 space-y-6">

      {/* Banner */}

      <PermitBanner

        type="air"

        title={`${inputs.siteName} — Air Permit Analysis`}

        subtitle={`${inputs.state} · ${totalMW} MW installed · ${inputs.turbines} turbines · ${inputs.hours.toLocaleString()} hr/yr`}

        badges={bannerBadges.map(b => (

          <PathwayBadge key={b.label} {...b} />

        ))}

      />

      {/* Metric Cards */}

      <div className="grid grid-cols-2 md:grid-cols-4 gap-4">

        {airMetrics.map(m => (

          <MetricCard key={m.label} {...m} />

        ))}

      </div>

      {/* Pathway + Modules */}

      <div className="grid md:grid-cols-2 gap-6">

        <PathwayDeterminationCard

          type="air"

          accentColor={c.accent}

          title="Permit Pathway Determination"

          items={pathwayItems}

        />

        <PermitModuleAccordion

          type="air"

          title="Air Permit Modules — Deliverables"

          modules={AIR_MODULES}

          accentColor={c.accent}

        />

      </div>

    </div>

  );

}
