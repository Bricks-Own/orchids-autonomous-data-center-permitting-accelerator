Part 1 — Document preview text is invisible (white text on white background)
In permitOS/src/components/DocumentPreviewModal.jsx, the document "paper" container (<div className="flex-1 overflow-y-auto bg-white">, line 212 onward) has a literal white background — correct, it's meant to look like a printed page. The result: document body text is nearly invisible, white-on-white.

A printed document should always look like paper regardless of the app's dark theme — it shouldn't inherit dark-mode tokens at all. Fix: inside the white paper container specifically (both the smaller inline preview around line 30 and the full preview from line 212 onward), replace every theme-relative color class with explicit fixed dark colors meant for a light page:

text-muted-foreground, text-muted-foreground/70, text-muted-foreground/50 → text-gray-600, text-gray-500, text-gray-400 (pick the closest visual weight)
bg-muted → bg-gray-100
border-border → border-gray-300
text-foreground/90 → text-gray-800

Part 2 — There is no real PDF generation at all
Both "Download" (single document, DocumentPreviewModal.jsx handleDownloadText) and "Download Full Package" (DocumentFactory.jsx handleDownloadAll) currently just create a Blob with type: 'text/plain' and download it as a .txt file with ASCII-art box-drawing characters (═, ║, ╔) standing in for formatting — there is no PDF anywhere in this app. That's what "PDF fucked up" actually is: there isn't one.

Add real PDF generation using jspdf (add it as a dependency — npm install jspdf). Each document already has a clean structure from generateDocument(): { title, docNum, sections: [{ heading, content }] }. Build a shared helper, e.g. permitOS/src/utils/pdfExport.js:

import jsPDF from 'jspdf';

export function generateDocPdf(doc, siteName) {
  const pdf = new jsPDF({ unit: 'pt', format: 'letter' });
  const marginX = 54;
  let y = 60;
  const pageHeight = pdf.internal.pageSize.getHeight();
  const maxWidth = pdf.internal.pageSize.getWidth() - marginX * 2;

  pdf.setFont('helvetica', 'bold');
  pdf.setFontSize(16);
  pdf.text(doc.title, marginX, y);
  y += 20;
  pdf.setFont('helvetica', 'normal');
  pdf.setFontSize(10);
  pdf.text(`Document No.: ${doc.docNum}  |  Facility: ${siteName}`, marginX, y);
  y += 10;
  pdf.setDrawColor(200);
  pdf.line(marginX, y, marginX + maxWidth, y);
  y += 20;

  doc.sections.forEach(sec => {
    if (y > pageHeight - 80) { pdf.addPage(); y = 60; }
    pdf.setFont('helvetica', 'bold');
    pdf.setFontSize(11);
    pdf.text(sec.heading, marginX, y);
    y += 16;
    pdf.setFont('helvetica', 'normal');
    pdf.setFontSize(9);
    const lines = pdf.splitTextToSize(sec.content, maxWidth);
    lines.forEach(line => {
      if (y > pageHeight - 40) { pdf.addPage(); y = 60; }
      pdf.text(line, marginX, y);
      y += 12;
    });
    y += 12;
  });

  return pdf;
}

export function downloadDocAsPdf(doc, siteName) {
  const pdf = generateDocPdf(doc, siteName);
  pdf.save(`${doc.docNum}_${doc.title.replace(/[^a-z0-9]+/gi, '_')}.pdf`);
}

export function downloadPackageAsPdf(docs, siteName) {
  const pdf = new jsPDF({ unit: 'pt', format: 'letter' });
  docs.forEach((doc, i) => {
    if (i > 0) pdf.addPage();
    // reuse the same per-doc rendering logic — refactor generateDocPdf to accept an existing pdf instance and starting y instead of always creating a new one, so this can append pages to a single combined PDF
  });
  pdf.save('Brick_PermitOS_Document_Package.pdf');
}
